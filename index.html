function runMatching() { let members = []; const num = parseInt(document.getElementById('numPeople').value); for(let i=1; i<=num; i++) { members.push({ name: document.getElementById(`name${i}`).value, sex: document.querySelector(`input[name="sex${i}"]:checked`).value, skill: parseInt(document.getElementById(`skill${i}`).value) }); } // [알고리즘] 여성 상위 6명 실력 보정 let sorted = [...members].sort((a,b) => b.skill - a.skill); members.forEach(m => { let rank = sorted.indexOf(m); if(rank < 6 && m.sex === 'F') m.skill += 2; }); // 팀 밸런싱을 위해 실력순 정렬 members.sort((a,b) => b.skill - a.skill); // 경기 스케줄 생성 const mGames = parseInt(document.getElementById('gameM').value); const mixGames = parseInt(document.getElementById('gameMix').value); const fGames = parseInt(document.getElementById('gameF').value); const scheduleTypes = Array(mGames).fill('남복') .concat(Array(mixGames).fill('혼복')) .concat(Array(fGames).fill('여복')); let result = "--- 최종 대진표 ---\n\n"; // 2경기씩 묶어서 세트로 처리 (1세트 = 2경기) for(let i = 0; i < scheduleTypes.length; i += 2) { let gameA = scheduleTypes[i]; let gameB = (i + 1 < scheduleTypes.length) ? scheduleTypes[i+1] : "휴식"; result += `[${(i/2)+1}세트]\n`; result += `1게임-A코트 (${gameA}): [${members[0].name}, ${members[7].name}] vs [${members[1].name}, ${members[6].name}]\n`; result += `1게임-B코트 (${gameB}): [${members[2].name}, ${members[5].name}] vs [${members[3].name}, ${members[4].name}]\n\n`; } document.getElementById('step3').style.display = 'none'; document.getElementById('resultCard').style.display = 'block'; document.getElementById('resultArea').innerText = result; }
